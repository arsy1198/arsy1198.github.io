data("instacart")

instacart =
  instacart  |> 
  select(
    order_dow, order_hour_of_day, product_name, aisle, department
  ) |> 
  add_count(aisle, name = "aisle_count") |> 
  filter(aisle_count > 10000)

set.seed(123)
instacart_sample <- instacart |> 
  sample_n(9000)

Chart A

instacart_sample |> 
  count(department, aisle, name = "aisle_count") |> 
  plot_ly(
    x = ~department, y = ~aisle_count, color = ~department,
    type = "box", colors = "viridis"
  )
